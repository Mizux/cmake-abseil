cmake_minimum_required(VERSION 3.5)

# simplify variable expansion
cmake_policy(SET CMP0053 NEW)
cmake_policy(SET CMP0010 NEW)

project(abseil-download NONE)

include(ExternalProject)
ExternalProject_Add(abseil_project
	GIT_REPOSITORY https://github.com/abseil/abseil-cpp
	GIT_TAG "master"
	SOURCE_DIR "${CMAKE_BINARY_DIR}/abseil-src"
	BINARY_DIR "${CMAKE_BINARY_DIR}/abseil-build"
	UPDATE_COMMAND ""
	#PATCH_COMMAND git apply "${CMAKE_SOURCE_DIR}/patches/abseil.patch"
	CONFIGURE_COMMAND ""
	BUILD_COMMAND ""
	INSTALL_COMMAND ""
	TEST_COMMAND ""
	LOG_DOWNLOAD ON
)
