cmake_minimum_required(VERSION 3.5)

# simplify variable expansion
cmake_policy(SET CMP0053 NEW)
cmake_policy(SET CMP0010 NEW)

project(cctz-download NONE)

include(ExternalProject)
ExternalProject_Add(cctz_project
	GIT_REPOSITORY https://github.com/google/cctz
	GIT_TAG "master"
	SOURCE_DIR "${CMAKE_BINARY_DIR}/cctz-src"
	BINARY_DIR "${CMAKE_BINARY_DIR}/cctz-build"
	UPDATE_COMMAND ""
	#PATCH_COMMAND git apply "${CMAKE_SOURCE_DIR}/patches/cctz.patch"
	CONFIGURE_COMMAND ""
	BUILD_COMMAND ""
	INSTALL_COMMAND ""
	TEST_COMMAND ""
	LOG_DOWNLOAD ON
)
